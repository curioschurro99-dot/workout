/* ---------------- TODAY ---------------- */
function todayKey(){ return new Date().toISOString().slice(0,10); }

function renderToday(){
  const prog = activeProgram();
  const stripEl = document.getElementById('day-strip');
  const cardEl = document.getElementById('today-card');
  const titleEl = document.getElementById('today-title');
  const descEl = document.getElementById('today-desc');

  if(!prog){
    stripEl.innerHTML='';
    cardEl.innerHTML = '<div class="empty">No active program yet. Go to <b>Programs</b> to start one.</div>';
    titleEl.textContent='Today';
    descEl.textContent='Pick an active program to see your workout for today.';
    return;
  }

  titleEl.textContent = prog.name;
  descEl.textContent = prog.desc || '';

  stripEl.innerHTML='';
  prog.days.forEach((d,i)=>{
    const pill = document.createElement('div');
    pill.className = 'day-pill' + (i===DB.activeDayIndex ? ' current':'');
    pill.textContent = d.label;
    pill.style.cursor='pointer';
    pill.addEventListener('click', ()=>{ DB.activeDayIndex = i; saveDB(); renderToday(); });
    stripEl.appendChild(pill);
  });

  const day = prog.days[DB.activeDayIndex];
  if(!day || day.exercises.length===0){
    cardEl.innerHTML = `<h3 style="color:var(--brass);margin-bottom:8px;">${day?day.focus:''}</h3><div class="empty">Rest day. Recovery is part of the plan — no exercises logged today.</div>`;
    return;
  }

  const checkKey = `${prog.id}_${DB.activeDayIndex}_${todayKey()}`;
  const checks = DB.checks[checkKey] || {};

  let html = `<h3 style="color:var(--brass);margin-bottom:4px;">${day.focus}</h3>`;
  day.exercises.forEach((pe,i)=>{
    const ex = exById(pe.exId);
    if(!ex) return;
    const isChecked = checks[i];
    html += `<div class="exercise-row">
      <div class="check ${isChecked?'checked':''}" data-idx="${i}">${isChecked?'✓':''}</div>
      <div class="ex-info">
        <div class="name">${ex.name}</div>
        <div class="meta">${ex.muscle} · ${pe.sets} sets × ${pe.reps} ${ex.note?('· '+ex.note):''}</div>
      </div>
    </div>`;
  });
  html += `<div style="margin-top:16px;"><button class="btn" id="finish-day-btn">Log Session Complete</button></div>`;
  cardEl.innerHTML = html;

  cardEl.querySelectorAll('.check').forEach(c=>{
    c.addEventListener('click', ()=>{
      const idx = c.dataset.idx;
      checks[idx] = !checks[idx];
      DB.checks[checkKey] = checks;
      saveDB();
      renderToday();
    });
  });
  document.getElementById('finish-day-btn').addEventListener('click', ()=>{
    DB.logs.push({date: todayKey(), programId: prog.id, dayLabel: day.label, focus: day.focus, exCount: day.exercises.length});
    saveDB();
    alert('Session logged for '+day.label+'. Nice work.');
    renderProgress();
  });
}

let timerState = {round:1, phase:'work', remaining:40, running:false, interval:null};
function fmtTime(s){ const m=Math.floor(s/60); const sec=s%60; return String(m).padStart(2,'0')+':'+String(sec).padStart(2,'0'); }

function updateTimerDisplay(){
  const disp = document.getElementById('timer-display');
  const label = document.getElementById('timer-phase-label');
  disp.textContent = fmtTime(timerState.remaining);
  disp.className = 'timer-display ' + (timerState.phase==='work'?'work':'rest');
  const rounds = parseInt(document.getElementById('t-rounds').value)||1;
  label.textContent = `${timerState.phase==='work'?'Work':'Rest'} — Round ${timerState.round}/${rounds}`;
}

function beep(){
  try{
    const ctx = new (window.AudioContext||window.webkitAudioContext)();
    const osc = ctx.createOscillator();
    const gain = ctx.createGain();
    osc.connect(gain); gain.connect(ctx.destination);
    osc.frequency.value = 880;
    gain.gain.setValueAtTime(0.15, ctx.currentTime);
    osc.start();
    osc.stop(ctx.currentTime+0.15);
  }catch(e){}
}

document.getElementById('t-start').addEventListener('click', ()=>{
  if(timerState.running) return;
  if(timerState.remaining<=0 || timerState.phase==='idle'){
    timerState.phase='work';
    timerState.round=1;
    timerState.remaining=parseInt(document.getElementById('t-work').value)||40;
  }
  timerState.running=true;
  updateTimerDisplay();
  timerState.interval = setInterval(()=>{
    timerState.remaining--;
    if(timerState.remaining<0){
      beep();
      const rounds = parseInt(document.getElementById('t-rounds').value)||1;
      if(timerState.phase==='work'){
        timerState.phase='rest';
        timerState.remaining = parseInt(document.getElementById('t-rest').value)||20;
      } else {
        if(timerState.round>=rounds){
          clearInterval(timerState.interval);
          timerState.running=false;
          document.getElementById('timer-phase-label').textContent='Complete';
          document.getElementById('timer-display').textContent='DONE';
          return;
        }
        timerState.round++;
        timerState.phase='work';
        timerState.remaining = parseInt(document.getElementById('t-work').value)||40;
      }
    }
    updateTimerDisplay();
  },1000);
});
document.getElementById('t-pause').addEventListener('click', ()=>{
  clearInterval(timerState.interval);
  timerState.running=false;
});
document.getElementById('t-reset').addEventListener('click', ()=>{
  clearInterval(timerState.interval);
  timerState.running=false;
  timerState.round=1; timerState.phase='work';
  timerState.remaining=parseInt(document.getElementById('t-work').value)||40;
  updateTimerDisplay();
});
updateTimerDisplay();

function renderProgress(){
  const logs = DB.logs;
  document.getElementById('stat-total').textContent = logs.length;

  const oneWeekAgo = new Date(); oneWeekAgo.setDate(oneWeekAgo.getDate()-7);
  const weekCount = logs.filter(l=>new Date(l.date) >= oneWeekAgo).length;
  document.getElementById('stat-week').textContent = weekCount;

  // streak: consecutive days with a log, counting back from today
  const dateSet = new Set(logs.map(l=>l.date));
  let streak=0; let d=new Date();
  while(dateSet.has(d.toISOString().slice(0,10))){ streak++; d.setDate(d.getDate()-1); }
  document.getElementById('stat-streak').textContent = streak;

  const goalEl = document.getElementById('goal-display');
  const g = DB.goal;
  const pct = g.weeklyTarget ? Math.min(100, Math.round((weekCount/g.weeklyTarget)*100)) : 0;
  goalEl.innerHTML = `
    <div style="font-size:13px;color:var(--bone-dim);margin-bottom:8px;">
      ${g.targetWeight ? `Target weight: <b style="color:var(--bone);">${g.targetWeight}kg</b> · ` : ''}
      Weekly target: <b style="color:var(--bone);">${g.weeklyTarget} sessions</b>
      ${g.notes ? `<div style="margin-top:6px;">${g.notes}</div>` : ''}
    </div>
    <div class="goal-bar-bg"><div class="goal-bar-fill" style="width:${pct}%;"></div></div>
    <div style="font-size:11px;color:var(--bone-dim);margin-top:4px;">${weekCount}/${g.weeklyTarget} sessions this week</div>
  `;

  const histEl = document.getElementById('history-list');
  if(logs.length===0){ histEl.innerHTML = '<div class="empty">No sessions logged yet.</div>'; }

  }

  if(!('speechSynthesis' in window)) return;

/* YouTube video embed */
function extractYoutubeId(url){
  if(!url) return null;
  const patterns = [
    /(?:youtube\.com\/watch\?v=|youtu\.be\/|youtube\.com\/embed\/|youtube\.com\/shorts\/)([a-zA-Z0-9_-]{11})/
  ];
  for(const p of patterns){ const m = url.match(p); if(m) return m[1]; }
  return null;
}
function renderVideoEmbed(){
  const wrap = document.getElementById('video-embed-wrap');
  document.getElementById('video-url-input').value = DB.videoUrl || '';
  const id = extractYoutubeId(DB.videoUrl);
  if(id){
    wrap.innerHTML = `<div class="video-embed"><iframe src="https://www.youtube.com/embed/${id}" title="Instructor video" allow="accelerometer; autoplay; clipboard-write; encrypted-media; gyroscope; picture-in-picture" allowfullscreen></iframe></div>`;
  } else {
    wrap.innerHTML = DB.videoUrl ? '<div class="empty">Could not read that link — paste a standard YouTube URL.</div>' : '';
  }
}
document.getElementById('video-save-btn').addEventListener('click', ()=>{
  DB.videoUrl = document.getElementById('video-url-input').value.trim();
  saveDB();
  renderVideoEmbed();
});
document.getElementById('video-clear-btn').addEventListener('click', ()=>{
  DB.videoUrl = '';
  saveDB();
  renderVideoEmbed();
});

function startSequence(){
  document.getElementById('yoga-browse').style.display = 'none';
  document.getElementById('yoga-sequence').style.display = 'block';
  renderVideoEmbed();
  goToPose(0);
}

function goToPose(idx){
  clearInterval(seqState.interval);
  speechSynthesis.cancel();
  if(idx < 0) idx = 0;
  if(idx >= SEQUENCE.length){
    document.getElementById('seq-name').textContent = 'Flow Complete';
    document.getElementById('seq-sans').textContent = '';
    document.getElementById('seq-cue').textContent = "Nice work — that's the full 26-posture flow. Rest in Savasana as long as you like.";
    document.getElementById('seq-timer').textContent = '';
    document.getElementById('seq-svg').innerHTML = SHAPES.corpse;
    document.getElementById('seq-progress').textContent = `Done — ${SEQUENCE.length}/${SEQUENCE.length}`;
    document.getElementById('seq-bar-fill').style.width = '100%';
    speak("That's the full flow. Rest here in Savasana for as long as you like.");
    return;
  }
  seqState.i = idx;
  const p = SEQUENCE[idx];
  seqState.remaining = p.hold;
  seqState.running = true;
  document.getElementById('seq-pause').textContent = 'Pause';
  document.getElementById('seq-svg').innerHTML = SHAPES[p.shape];
  document.getElementById('seq-name').textContent = p.name;
  document.getElementById('seq-sans').textContent = p.sanskrit;
  document.getElementById('seq-cue').textContent = p.cue;
  document.getElementById('seq-progress').textContent = `Posture ${idx+1}/${SEQUENCE.length}`;
  document.getElementById('seq-bar-fill').style.width = `${Math.round((idx/SEQUENCE.length)*100)}%`;
  document.getElementById('seq-timer').textContent = seqState.remaining;
  speak(`${p.name}. ${p.cue}`);
  runSeqTimer();
}

function runSeqTimer(){
  clearInterval(seqState.interval);
  seqState.interval = setInterval(()=>{
    seqState.remaining--;
    document.getElementById('seq-timer').textContent = Math.max(0,seqState.remaining);
    if(seqState.remaining === 3 && document.getElementById('voice-coach-toggle').checked){
      const u = new SpeechSynthesisUtterance('almost there');
      u.rate = 1; speechSynthesis.speak(u);
    }
    if(seqState.remaining <= 0){
      beep();
      goToPose(seqState.i+1);
    }
  }, 1000);
}
